# 0 Country made vassal or tributary; called by on_becoming_subject
country_event = {
	id = exalted_subjects.0
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		exists = FROM
		FROM = { exalted_is_valid_as_vassal = yes }
	}

	immediate = {
		if = {
			limit = {
				OR = {
					is_subject_type = vassal
					is_exalted_vassal = yes
				}
			}

			log = "exalted_subjects.0: [This.GetName] has been vassalized by [From.GetName]"
			if = {
				limit = {
					is_gestalt = no
					is_megacorp = no
				}
				FROM = {
					country_event = { id = exalted_subjects.5 days = 3 }
				}
			}
			else_if = {
				limit = {
					is_gestalt = yes
					is_machine_empire = yes
				}
				FROM = {
					country_event = { id = exalted_subjects.7 days = 3 }
				}
			}
			else_if = {
				limit = {
					is_gestalt = yes
					is_hive_empire = yes
				}
				FROM = {
					country_event = { id = exalted_subjects.8 days = 3 }
				}
			}
		}
		else_if = {
			limit = {
				is_subject_type = tributary
			}

			log = "exalted_subjects.0: [This.GetName] has been made a tributary by [From.GetName]"
			FROM = {
				country_event = { id = exalted_subjects.6 days = 3 }
			}
		}
	}
}

# 1 Country given AI bonus; called by on_becoming subject
country_event = {
	id = exalted_subjects.1
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		NOT = { has_country_flag = exalted_domination_difficulty_bonus }
		is_ai = yes
		exists = FROM
		FROM = { is_ai = no }
	}

	immediate = {
		set_country_flag = exalted_domination_difficulty_bonus

		if = {
			limit = { 
				OR = {
					is_difficulty = 0
					is_difficulty = 1
					is_difficulty = 2
				}
			}

			log = "exalted_subjects.1: [This.GetName] granted Captain difficulty bonus"
			add_modifier = { modifier = difficulty_captain }
		}
		else_if = {
			limit = {
				is_difficulty = 3
			}

			log = "exalted_subjects.1: [This.GetName] granted Commodore difficulty bonus"
			add_modifier = { modifier = difficulty_commodore }
		}
		else_if = {
			limit = {
				is_difficulty = 4
			}

			log = "exalted_subjects.1: [This.GetName] granted Admiral difficulty bonus"
			add_modifier = { modifier = difficulty_admiral }
		}
		else_if = {
			limit = {
				is_difficulty = 5
			}

			log = "exalted_subjects.1: [This.GetName] granted Grant Admiral difficulty bonus"
			add_modifier = { modifier = difficulty_grand_admiral }
		}
		else = {
			log = "exalted_subjects.1: [This.GetName] granted no difficulty bonus"
		}
	}
}

# 2 Check for vanilla vassals; called by on_monthly_pulse_country
# This is necessary because there are no actions for protectorates becoming vassals
country_event = {
	id = exalted_subjects.2
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		NOR = {
			has_country_flag = exalted_advanced_tributary_offered
			has_country_flag = exalted_advanced_vassal_offered
		}
		OR = { 
			is_subject_type = tributary
			AND = {
				is_subject_type = vassal
				is_megacorp = no
			}
		}
		exists = OVERLORD
		OVERLORD = { is_country_type = default }
	}

	immediate = {
		if = {
			limit = {
				is_subject_type = vassal
			}

			if = {
				limit = {
					is_gestalt = no
					is_megacorp = no
				}

				log = "exalted_subjects.2: [This.GetName] is a vanilla vassal; offer chance to change"
				OVERLORD = {
					country_event = { id = exalted_subjects.5 }
				}
			}
			else_if = {
				limit = {
					is_gestalt = yes
					is_machine_empire = yes
				}

				log = "exalted_subjects.2: [This.GetName] is a vanilla machine empire vassal; offer chance to change"
				OVERLORD = {
					country_event = { id = exalted_subjects.7 }
				}
			}
			else_if = {
				limit = {
					is_gestalt = yes
					is_hive_empire = yes
				}

				log = "exalted_subjects.2: [This.GetName] is a vanilla hive mind vassal; offer chance to change"
				OVERLORD = {
					country_event = { id = exalted_subjects.8 }
				}
			}
			else = {
				log = "exalted_subjects.2: [This.GetName] is a not supported empire type for advanced vassals"
			}
		}
		else_if = {
			limit = {
				is_subject_type = tributary
			}
	
			log = "exalted_subjects.2: [This.GetName] is a vanilla tributary; offer chance to change"
			OVERLORD = {
				country_event = { id = exalted_subjects.6 }
			}
		}
	}
}

# 11 Becoming Dominion
country_event = {
	id = exalted_subjects.11
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		NOT = { has_country_flag = exalted_dominion }
		is_subject_type = exalted_subject_dominion
	}

	immediate = {
		log = "exalted_subjects.11: [This.GetName] is becoming Dominion to [From.GetName]"

		# Set some pops to Authoritarian
		while = {
			count = @NumPopsShiftedSome
			random_owned_pop = {
				limit = {
					NOR = {
						has_ethic = ethic_authoritarian
						has_ethic = ethic_spiritualist
					}
				}

				clear_ethos = yes
				pop_force_add_ethic = ethic_authoritarian
			}
		}

		# Set some pops to Spiritualist
		while = {
			count = @NumPopsShiftedSome
			random_owned_pop = {
				limit = {
					NOR = {
						has_ethic = ethic_authoritarian
						has_ethic = ethic_spiritualist
					}
				}

				clear_ethos = yes
				pop_force_add_ethic = ethic_spiritualist
			}
		}

		# Make government some form of authoritarian spiritualist
		if = {
			limit = {
				has_ethic = ethic_fanatic_authoritarian
			}

			log = "exalted_subjects.11: [This.GetName] is fanatic authoritarian"

			if = {
				limit = {
					is_spiritualist = no
				}
				clear_ethos = yes
				country_add_ethic = ethic_fanatic_authoritarian
				country_add_ethic = ethic_spiritualist

				log = "exalted_subjects.11: [This.GetName] made spiritualist"
			}
			else = {
				log = "exalted_subjects.11: [This.GetName] is spiritualist"
			}
		}
		else_if = {
			limit = {
				has_ethic = ethic_fanatic_spiritualist
			}

			log = "exalted_subjects.11: [This.GetName] is fanatic spiritualist"
			if = {
				limit = {
					is_authoritarian = no
				}
				clear_ethos = yes
				country_add_ethic = ethic_fanatic_spiritualist
				country_add_ethic = ethic_authoritarian

				log = "exalted_subjects.11: [This.GetName] made authoritarian"
			}
			else = {
				log = "exalted_subjects.11: [This.GetName] is authoritarian"
			}
		}
		else = {
			random_list = {
				50 = {
					clear_ethos = yes
					country_add_ethic = ethic_fanatic_authoritarian
					country_add_ethic = ethic_spiritualist
					log = "exalted_subjects.11: [This.GetName] made spiritualist and fanatic authoritarian"
				}
				50 = {
					clear_ethos = yes
					country_add_ethic = ethic_fanatic_spiritualist
					country_add_ethic = ethic_authoritarian
					log = "exalted_subjects.11: [This.GetName] made authoritarian and fanatic spiritualist"
				}
			}
		}

		set_country_flag = exalted_dominion
		change_government = {
			authority = auth_imperial
			civics = {
				civic = civic_imperial_cult
				civic = random
			}
		}

		# Note Ruler/Heir
		overlord = {
			if = {
				limit = {
					exists = ruler
				}
				ruler = {
					save_event_target_as = exalted_domination_overlord_ruler
				}
			}
		}

		# Execute subject's ruler
		country_event = { id = exalted_subjects.110 days = 1 }
	}
}

# 181 House - New ruler appointed, execute vassal's current ruler and heir
country_event = {
	id = exalted_subjects.181
	title = exalted_subjects.181.name
	desc = exalted_subjects.181.desc
	picture = GFX_evt_signing
	is_triggered_only = yes

	trigger = {
		is_subject_type = exalted_subject_noble_house
		exists = event_target:exalted_domination_house_ruler
		exists = event_target:exalted_domination_main_species
	}

	immediate = {
		log = "exalted_subjects.181: [From.GetName] appoints new ruler [event_target:exalted_domination_house_ruler.GetName] to [This.GetName]"

		owner_species = {
			save_global_event_target_as = "exalted_subjects_house_original_species_@from"
		}
		
		# Set dominant species for new rulers and heirs
		change_dominant_species = {
			species = event_target:exalted_domination_main_species
		}

		# Replace some capital pops with Overlord's species
		capital_scope = {
			while = {
				count = @NumPopsShiftedSome
				random_owned_pop = {
					limit = {
						species = {
							NOT = { is_same_value = event_target:exalted_domination_main_species }
						}
					}

					kill_pop = yes

					prev = {
						create_pop = {
							species = event_target:exalted_domination_main_species
						}

						last_created_pop = {
							pop_force_add_ethic = ethic_authoritarian
						}
					}
				}
			}
		}

		if = {
			limit = { exists = ruler }
			
			ruler = {
				kill_leader = { show_notification = yes }
			}
		}

		if = {
			limit = { exists = heir }

			heir = {
				kill_leader = { show_notification = no }
			}
		}

		# set appointed leader as new ruler of house
		clone_leader = {
			target = event_target:exalted_domination_house_ruler
			class = ruler
		}
	}

	option = {
		name = exalted_subjects.181.choice
	}
}
